9 def OperationalAmplifier(Zi,Zd,Zo,G,Z0=50.):
10  """OperationalAmplifier
11  Operational Amplifier
12  @param Zi real or complex input impedance between each plus and minus input port
13  to ground.
14  @param Zd real or complex input impedance across the plus and minus inputs.
15  @param Zo real or complex output impedance.
16  @param G real or complex gain of the op-amp.
17  @param Z0 (optional) real or complex reference impedance (default is 50 Ohms).
18  @return the list of list s-parameter matrix of the s-parameters of a three port op-amp.
19  Port 1 is - input, port 2 is + input and port 3 is output
20  """
21  S11=((-2.*Zi-Zd)*Z0*Z0+Zi*Zi*Zd)/((2.*Zi+Zd)*Z0*Z0+(2.*Zi*Zd+2.*Zi*Zi)*Z0+Zi*Zi*Zd)
22  S12=(2.*Zi*Zi*Z0)/((2.*Zi+Zd)*Z0*Z0+(2.*Zi*Zd+2.*Zi*Zi)*Z0+Zi*Zi*Zd)
23  S32=2.*G*Zi*Zd*Z0/((Zo+Z0)*((2.*Zi+Zd)*Z0+Zi*Zd))
24  S33=(Zo-Z0)/(Zo+Z0)
25  return [[S11,S12,0.],[S12,S11,0.],[-S32,S32,S33]]
