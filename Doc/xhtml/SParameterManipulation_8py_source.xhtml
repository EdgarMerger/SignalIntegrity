24 class SParameterManipulation(object):
25  """Class for manipulations on s-parameters involving passivity, causality, etc."""
26  # locations where the largest singular value exceeds 1 are locations
27  # where there are passivity violations
28  def _LargestSingularValues(self):
29  return [linalg.svd(m,full_matrices=False,compute_uv=False)[0]
30  for m in self.m_d]
31  def EnforcePassivity(self,maxSingularValue=1.):
32  """Enforces passivity on the s-parameters.
33  @param maxSingularValue (optional) float maximum singular value allowed
34  Enforces passivity by clipping all singular values to a maximum value.
35 
36  the optional maximum value allows for adjusting devices with gain to a maximum
37  value. For passive devices, the maximum singular value is the default value of 1.
38  """
39  for n in range(len(self.m_d)):
40  (u,s,vh)=linalg.svd(self.m_d[n],full_matrices=1,compute_uv=1)
41  for si in range(len(s)):
42  s[si]=min(maxSingularValue,s[si])
43  self.m_d[n]=dot(u,dot(diag(s),vh)).tolist()
44  return self
45  def IsCausal(self,threshold=0.):
46  """Checks whether the s-parameters are causal.
47  @param threshold positive float threshold for causality detection
48  @return boolean True if the absolute value of all values in the impulse response
49  of each s-parameter before time zero are less
50  than the threshold provided otherwise returns False.
51 
52  This is checked by generating the impulse response corresponding to the frequency
53  response of each s-parameter from and to port combination.
54  """
55  for toPort in range(self.m_P):
56  for fromPort in range(self.m_P):
57  fr=self.FrequencyResponse(toPort,fromPort)
58  ir=fr.ImpulseResponse()
59  if ir is None: return False
60  else:
61  t=ir.td
62  Ts=1./ir.td.Fs
63  for k in range(len(t)):
64  if t[k]<=-Ts:
65  if abs(ir[k])>threshold:
66  return False
67  return True
68  def EnforceCausality(self):
69  """Enforces causality by setting all of the values before time zero in the impulse
70  responses of the s-parameters to zero."""
71  for toPort in range(self.m_P):
72  for fromPort in range(self.m_P):
73  fr=self.FrequencyResponse(toPort+1,fromPort+1)
74  ir=fr.ImpulseResponse()
75  if ir is not None:
76  t=ir.td
77  Ts=1./ir.td.Fs
78  for k in range(len(t)):
79  if t[k]<=-Ts:
80  ir[k]=0.
81  fr=ir.FrequencyResponse()
82  frv=fr.Response()
83  for n in range(len(frv)):
84  self.m_d[n][toPort][fromPort]=frv[n]
85  return self
86  def WaveletDenoise(self,threshold=0.0001):
87  """Denoises the s-parameters
88  @param threshold (optional) float threshold for the wavelets (defaults to 0.0001).
89  Denoises the s-parameter by computing the wavelet transform of the impulse response
90  for each s-parameter from and to port combination and keeping only the wavelets whoe
91  absolute value is above the threshold."""
92  from SignalIntegrity.Lib.Wavelets import WaveletDaubechies4
93  w=WaveletDaubechies4()
94  for toPort in range(self.m_P):
95  for fromPort in range(self.m_P):
96  fr=self.FrequencyResponse(toPort,fromPort)
97  ir=fr.ImpulseResponse()
98  if ir is not None:
99  irl=len(ir)
100  nirl=int(pow(2.,math.ceil(math.log(float(irl))/math.log(2.))))
101  ir=ir._Pad(nirl)
102  y=ir.Values()
103  Y=w.DWT(y)
104  Y=[0. if abs(Yv) <= threshold else Yv for Yv in Y]
105  y=w.IDWT(Y)
106  #ir.x=y
107  # @todo get rid of this hack
108  ir.__init__(ir.TimeDescriptor(),y)
109  ir=ir._Pad(irl)
110  fr=ir.FrequencyResponse()
111  frv=fr.Response()
112  for n in range(len(frv)):
113  self.m_d[n][toPort][fromPort]=frv[n]
