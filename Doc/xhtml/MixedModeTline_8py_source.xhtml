20 class MixedModeTLine(SParameters):
21  """s-parameters of a mixed-mode, balanced transmission line"""
22  def __init__(self,f,Zd,Td,Zc,Tc,Z0=50.):
23  """Constructor
24  The port numbering is ports 1 and 2 are the plus and minus terminals on
25  one side of the line and ports 3 and 4 are the plus and minus terminals
26  on the other side of the line.
27 
28  @param f list of float frequencies
29  @param Zd float or complex differential mode impedance
30  @param Td float differential (or odd-mode) electrical length (the
31  differential-mode propagation time.
32  @param Zc float or complex common-mode impedance
33  @param Tc float common (or even-mode) electrical length (the common-mode
34  propagation time.
35  @param Z0 (optional) float or complex reference impedance (defaults to 50 Ohms).
36  @note The differential mode impedance is twice the odd-mode impedance.\n
37  The common-mode impedance is half the even-mode impedance.\n
38  @note The model is appropriate for a balanced transmission line.\n
39  @note The s-parameters provided are single-ended.
40  @note This device builds a model internally and only solves the model
41  on each access to __getitem__().
42  """
43  import SignalIntegrity.SParameters.Devices as dev
44  from SignalIntegrity.Devices.MixedModeConverter import MixedModeConverter
45  from SignalIntegrity.SystemDescriptions.SystemSParametersNumeric import SystemSParametersNumeric
46  self.m_sspn=SystemSParametersNumeric()
47  self.m_spdl=[]
48  self.m_sspn.AddDevice('MM1',4,MixedModeConverter())
49  self.m_sspn.AddDevice('MM2',4,MixedModeConverter())
50  self.m_sspn.AddDevice('D',2), self.m_spdl.append(('D',dev.TLineLossless(f,2,Zd/2.,Td,Z0)))
51  self.m_sspn.AddDevice('C',2), self.m_spdl.append(('C',dev.TLineLossless(f,2,Zc*2.,Tc,Z0)))
52  self.m_sspn.ConnectDevicePort('MM1',3,'D',1)
53  self.m_sspn.ConnectDevicePort('MM2',3,'D',2)
54  self.m_sspn.ConnectDevicePort('MM1',4,'C',1)
55  self.m_sspn.ConnectDevicePort('MM2',4,'C',2)
56  self.m_sspn.AddPort('MM1',1,1)
57  self.m_sspn.AddPort('MM1',2,2)
58  self.m_sspn.AddPort('MM2',1,3)
59  self.m_sspn.AddPort('MM2',2,4)
60  SParameters.__init__(self,f,None,Z0)
61  def __getitem__(self,n):
62  """overloads [n]
63  @return list of list s-parameter matrix for the nth frequency element
64  """
65  for ds in self.m_spdl: self.m_sspn.AssignSParameters(ds[0],ds[1][n])
66  return self.m_sspn.SParameters()
