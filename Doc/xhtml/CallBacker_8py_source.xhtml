13 class CallBacker(object):
14  """CallBacker
15 
16  To use callbacks, a class is derived from CallBacker.
17 
18  Either the constructor of the derived class or some other mechanism should
19  provide means for supplying the callback function, which is installed in a
20  call to __init__.
21 
22  The derived class, during some long operation, then calls the callback function
23  periodically. The callback function then has the opportunity to report its
24  progress. Returning False in the callback function should cause the operation
25  to abort.
26 
27  Here's a mini example derived class using CallBacker:
28 
29  \code{.py}
30  class ProcessingClass(Callbacker):
31  def __init__(self,callback=None):
32  ...
33  Callbacker.__init__(callback)
34  ...
35  def LongProcessingFunction(self):
36  for m in range(100):
37  ...
38  do some calculating.
39  ...
40  if not self.CallBack(m):
41  process was aborted
42  return
43  \endcode
44  """
45  def __init__(self,callback=None):
46  """Constructor
47  @param callback (optional, defaults to None) a callback function to call
48  The callback function should take a number as an argument, which represents
49  the progress."""
50  self.InstallCallback(callback)
51  def HasACallBack(self):
52  return not self.callback is None
53  def CallBack(self,progressPercent):
54  """This function is called periodically, which in turn calls any installed
55  callback function.
56  @param progressPercent the progress in percent
57  """
58  if not self.callback is None:
59  return self.callback(progressPercent)
60  else:
61  return True

66  The callback function should take a number as an argument, which represents
67  the progress.
68  """
69  self.callback = callback
