24 class ImpedanceProfile(list):
25  rhoLimit=0.99
26  ZLimit=10e3
27  def __init__(self,sp,sections,port):
28  list.__init__(self,[])
29  N = len(sp)-1
30  self.m_Td = 1./(4.*sp.f()[N])
31  self.m_Z0 = sp.m_Z0
32  fr=sp.FrequencyResponse(port,port)
33  self.m_fracD=fr._FractionalDelayTime()
34  fr=fr._DelayBy(-self.m_fracD)
35  S11 = fr.Values()
36  zn2 = [cmath.exp(-1j*2.*math.pi*n/N*1/2) for n in range(N+1)]
37  finished=False
38  rho=0.0
39  for _ in range(sections):
40  if finished:
41  self.append(rho)
42  continue
43  rho = 1/(2.*N)*(S11[0].real + S11[N].real +
44  sum([2.*S11[n].real for n in range(1,N)]))
45  rho=max(-self.rhoLimit,min(rho,self.rhoLimit))
46  self.append(rho)
47  if abs(rho)==self.rhoLimit:
48  finished=True
49  continue
50  rho2=rho*rho
51  S11=[(-S11[n]+S11[n]*rho2*zn2[n]-rho*zn2[n]+rho)/
52  (rho2+S11[n]*rho*zn2[n]-S11[n]*rho-zn2[n])
53  for n in range(N+1)]
54  def Z(self):
55  return [max(0.,min(self.m_Z0*(1+rho)/(1-rho),self.ZLimit))
56  for rho in self]
57  def DelaySection(self):
58  return self.m_Td
59  def SParameters(self,f):
60  N = len(f)-1
61  Gsp=[None for n in range(N+1)]
62  gamma=[1j*2.*math.pi*f[n]*self.m_Td for n in range(N+1)]
63  for n in range(N+1):
64  tacc=matrix([[1.,0.],[0.,1.]])
65  for m in range(len(self)):
66  tacc=tacc*matrix(S2T(IdealTransmissionLine(self[m],gamma[n])))
67  Gsp[n]=T2S(tacc.tolist())
68  sp = SParameters(f,Gsp,self.m_Z0)
69  return sp
