14 import cmath

17 def CZT(x,Fs,fs,fe,M,highSpeed=True):
18  """Provides M+1 points from fs to fe of the chirp z transform of the vector x
19  @param x an input vector (list) of real time-domain samples
20  @param Fs the sample rate of the the data samples in x
21  @param fs the start frequency for the result
22  @param fe the end frequency for the result
23  @param M the number of points in the result (-1)
24  @param highSpeed (optional, defaults to True) whether to use the high-speed,
25  but unreadable version. The readable, low-speed version is the definition
26  is used for testing.
27  @remark
28  Technically, the chirp-z transform can evaluate the z-transform at equi-angularly
29  spaced points on a somewhat arbitrary arc, but here, we use it to compute equi-angularly
30  spaced points on the rim of the unit circle, and as such, it is used simply to resample
31  data. Usually, we set fs=0 and the result is an M+1 point DFT of the waveform from 0 to
32  fs, without the same limitations imposed as the standard DFT.\n
33 
34  The equation for the czt is:
35 
36  \f$m\in 0 \ldots M\f$
37 
38  \f$X\left[m\right]=\sum_{k=0}^{K-1}x\left[k\right]\cdot e^{-j\cdot2\pi\cdot\frac{k\cdot m}
39  {M}\cdot F_{e}}\f$
40  """
41  M=int(M); K=len(x); fs=float(fs); Fs=float(Fs); fe=float(fe)
42  theta0=fs/Fs; phi0=(fe-fs)/Fs*1./M; A0=1.; W0=1.
43  A=A0*cmath.exp(1j*2.*math.pi*theta0)
44  W=W0*cmath.exp(1j*2.*math.pi*phi0)
45  if highSpeed:
46  L=pow(2,int(math.ceil(math.log(float(M+K),2.))))*2
47  v=[0 for l in range(L)]
48  for k in range(M+1): v[k]=pow(W,float(k*k)/2.)
49  for k in range(M+1,L-K+1): v[k]=0.
50  for k in range(L-K+1,L): v[k]=pow(W,(L-k)*(L-k)/2.)
51  V=fft.fft(v)
52  y = [0 for l in range(L)]
53  for k in range(K): y[k]=x[k]*pow(A,-k)*pow(W,-(k*k)/2.)
54  for k in range(K,L): y[k]=0
55  Y=fft.fft(y)
56  P=[Y[l]*V[l] for l in range(L)]
57  P=fft.ifft(P)
58  X = [P[m]/v[m] for m in range(M+1)]
59  else:
60  Zeta=[A*pow(W,m) for m in range(M+1)]
61  X=[]
62  for m in range(M+1):
63  Xm=0.
64  for k in range(K): Xm=Xm+x[k]*pow(Zeta[m],-k)
65  X.append(Xm)
66  return X
