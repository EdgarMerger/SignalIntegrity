32 def VoltageAmplifierFourPort(G,Zi,Zo,Z0=50.):
33  """VoltageAmplifierFourPort
34  Four port voltage amplifier
35  @param G float voltage gain
36  @param Zi float or complex input impedance
37  @param Zo float or complex output impedance
38  @param Z0 (optional) float reference impedance (defaults to 50 Ohms)
39  @return list of list s-parameter matrix for a voltage amplifier
40  @remark
41  The voltage sense element senses the voltage with the plus terminal at port 1
42  and and the minus terminal at port 2 shunted with impedance of Zi.\n
43  The voltage generated by the amplifier has the plus terminal at port 4 and
44  and the minus terminal at port 3 with Zo in series.\n
45  """
46  S11=Zi/(Zi+2.*Z0)
47  S12=2.*Z0/(Zi+2.*Z0)
48  S13=0
49  S14=0
50  S21=S12
51  S22=S11
52  S23=0
53  S24=0
54  S31=2.*Zi*Z0*G/((Zi+2.*Z0)*(Zo+2.*Z0))
55  S32=-S31
56  S33=Zo/(Zo+2.*Z0)
57  S34=2.*Z0/(Zo+2.*Z0)
58  S41=S32
59  S42=S31
60  S43=S34
61  S44=S33
62  return [[S11,S12,S13,S14],
63  [S21,S22,S23,S24],
64  [S31,S32,S33,S34],
65  [S41,S42,S43,S44]]

67 def VoltageAmplifierThreePort(G,Zi,Zo,Z0=50.):
68  """VoltageAmplifierThreePort
69  Three port voltage amplifier
70  @param G float voltage gain
71  @param Zi float or complex input impedance
72  @param Zo float or complex output impedance
73  @param Z0 (optional) float reference impedance (defaults to 50 Ohms)
74  @return list of list s-parameter matrix for a voltage amplifier
75  @remark
76  The three port voltage amplifier is the same as the four port voltage amplifier with
77  ports two and three connected together and exposed as a single port.\n
78  The voltage sense element senses the voltage with the plus terminal at port 1
79  and and the minus terminal at port 3 shunted with impedance of Zi.\n
80  The voltage generated by the amplifier has the plus terminal at port 2 and
81  and the minus terminal at port 3 with Zo in series.\n
82  """
83  D=-Zo*Zi-2.*Zo*Z0-2.*Zi*Z0-3.*Z0*Z0+G*Zi*Z0
84  S11=(-Zo*Zi-2.*Zi*Z0+Z0*Z0+G*Zi*Z0)/D
85  S12=-2.*Z0*Z0/D
86  S13=-2.*Z0*(Zo+Z0)/D
87  S21=-2.*Z0*(G*Zi+Z0)/D
88  S22=(Z0*Z0-2.*Zo*Z0+G*Zi*Z0-Zo*Zi)/D
89  S23=2.*Z0*(G*Zi-Zi-Z0)/D
90  S31=2.*Z0*(-Z0+G*Zi-Zo)/D
91  S32=-2.*Z0*(Zi+Z0)/D
92  S33=(-Zo*Zi+Z0*Z0-G*Zi*Z0)/D
93  return [[S11,S12,S13],
94  [S21,S22,S23],
95  [S31,S32,S33]]

97 def VoltageAmplifierTwoPort(G,Zi,Zo,Z0=50.):
98  """VoltageAmplifierTwoPort
99  Two port voltage amplifier
100  @param G float voltage gain
101  @param Zi float or complex input impedance
102  @param Zo float or complex output impedance
103  @param Z0 (optional) float reference impedance (defaults to 50 Ohms)
104  @return list of list s-parameter matrix for a voltage amplifier
105  @remark
106  The two port voltage amplifier is the same as the three port voltage amplifier with
107  port 3 grounded.\n
108  The voltage sense element senses the voltage to ground at port 1 shunted to ground
109  with impedance of Zi.\n
110  The voltage generated by the amplifier is at port 2 referenced to ground
111  with Zo in series with the output.\n
112  """
113  return [[(Zi-Z0)/(Zi+Z0),0.],[2.*G*Zi*Z0/((Zi+Z0)*(Zo+Z0)),(Zo-Z0)/(Zo+Z0)]]
