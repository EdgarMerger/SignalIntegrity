22 class Wavelet(object):
23  def __init__(self,h):
24  self.h=h
25  self.L=len(self.h)
26  self.g=[pow(-1.,l)*h[self.L-1-l] for l in range(self.L)]
27  def DWT(self,xi):
28  x=list(xi)
29  N=len(x)
30  B=self.intlog2(N)-self.intlog2(self.L)+1
31  for i in range(B):
32  X=list(x)
33  for k in range(N/2):
34  X[k]=sum([x[(2*k+l)%N]*self.h[l]
35  for l in range(self.L)])
36  X[k+N/2]=sum([x[(2*k+l+N-2)%N]*self.g[l]
37  for l in range(self.L)])
38  x=list(X)
39  N=N/2
40  return X

42  def IDWT(self,XI):
43  X=list(XI)
44  N=len(X)
45  B=self.intlog2(N)-self.intlog2(self.L)+1
46  N=self.L
47  for i in range(B):
48  x=list(X)
49  for k in range(N/2):
50  x[2*k]=sum([self.h[2*l]*X[(k-l+(N/2))%(N/2)]+
51  self.g[2*l]*X[(k+1-l+(N/2))%(N/2)+(N/2)]
52  for l in range(self.L/2)])
53  x[2*k+1]=sum([self.h[2*l+1]*X[(k-l+(N/2))%(N/2)]+
54  self.g[2*l+1]*X[(k+1-l+(N/2))%(N/2)+(N/2)]
55  for l in range(self.L/2)])
56  X=list(x)
57  N=2*N
58  return x
