35  def LaTeXTransferMatrix(self):
36  """Calculates and stores internally a symbolic representation
37  of the transfer matrix in LaTeX.
38  @return self
39  @see Symbolic
40  """
41  self.Check()
42  self._LaTeXSi()
43  numMeas=len(self.pMeasurementList)
44  vemsi = MatrixMultiply(
45  self.VoltageExtractionMatrix(self.pMeasurementList), self.SIPrime(True))
46  oneElementVemsi = False
47  if len(vemsi) == 1:
48  if len(vemsi[0]) == 1: oneElementVemsi = True
49  vemsi = Matrix2LaTeX(vemsi, self._SmallMatrix())
50  veosi = MatrixMultiply(
51  self.VoltageExtractionMatrix(self.pOutputList), self.SIPrime(True))
52  oneElementVeosi = False
53  if len(veosi) == 1:
54  if len(veosi[0]) == 1: oneElementVeosi = True
55  veosi = Matrix2LaTeX(veosi, self._SmallMatrix())
56  if self.pStimDef is None:
57  numDeg=len(self.SIPrime(True)[0])
58  inverse='^{-1}' if numDeg==numMeas else '^\\dagger'
59  if oneElementVemsi and oneElementVeosi:
60  line = '\\frac{ '+veosi+' }{ '+vemsi+' } '
61  elif oneElementVemsi and not oneElementVeosi:
62  line = veosi+' \\frac{1}{ '+vemsi+' } '
63  elif not oneElementVemsi and oneElementVeosi:
64  line = '\\left( '+veosi+'\\right)\\cdot '+vemsi+inverse
65  elif not oneElementVemsi and not oneElementVeosi:
66  line = veosi+'\\cdot '+vemsi+inverse
67  else:
68  numDeg=len(self.pStimDef[0])
69  inverse='^{-1}' if numDeg==numMeas else '^\\dagger'
70  D = Matrix2LaTeX(self.pStimDef, self._SmallMatrix())
71  fveosi='\\left( '+veosi+'\\right)' if oneElementVeosi else veosi
72  fvemsi='\\left( '+vemsi+'\\right)' if oneElementVemsi else vemsi
73  line = '\\left[ '+fveosi+'\\cdot '+D+'\\right]\\cdot'+\
74  '\\left[ '+fvemsi+'\\cdot '+D+'\\right]'+inverse
75  if len(self.pMeasurementList) == 1 and len(self.pOutputList) == 1: H = 'H'
76  else: H = '\\mathbf{H}'
77  return self._AddEq(H+' = '+line)
