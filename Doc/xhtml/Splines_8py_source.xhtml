22  def __init__(self,x,y):
23  N=len(x)
24  r = range(0,N-1)
25  h = [x[i+1]-x[i] for i in r]
26  b = [(y[i+1]-y[i])/h[i] for i in r]
27  u = [0,2.*(h[0]+h[1])]
28  v = [0,6.*(b[1]-b[0])]
29  for i in range(2,N-1):
30  u.append(2.*(h[i]+h[i-1])-h[i-1]*h[i-1]/u[i-1])
31  v.append(6.*(b[i]-b[i-1])-h[i-1]*v[i-1]/u[i-1])
32  zi = range(1,N-1)
33  zi.reverse()
34  z = [0 for i in range(N)]
35  for i in zi:
36  z[i]=(v[i]-h[i]*z[i+1])/u[i]
37  r=range(0,N-1)
38  A = [y[i] for i in r]
39  B = [-h[i]*z[i+1]/6.-h[i]*z[i]/3.+b[i] for i in r]
40  C = [z[i]/2. for i in r]
41  D = [1./6./h[i]*(z[i+1]-z[i]) for i in r]
42  self.m_t = [x[i] for i in r]
43  P=[]
44  #these are the actual polynomial coefficients for P[0]+P[1]*x+P[2]*x**2+P[3]*x**3
45  for i in r:
46  Pi=[]
47  Pi.append(A[i]+x[i]*(-B[i]+x[i]*(C[i]-x[i]*D[i])))
48  Pi.append(B[i]+x[i]*(-2.*C[i]+x[i]*3.*D[i]))
49  Pi.append(C[i]-3.*D[i]*x[i])
50  Pi.append(D[i])
51  P.append(Pi)
52  self.m_A=[]
53  #these are the polynomial coefficients for A[0]+A[1]*(x-t)+A[2]*(x-t)**2+A[3]*(x-t)**3
54  #where t is the lower boundary of the interval
55  for i in r:
56  Ai=[]
57  Ai.append(P[i][0]+x[i]*(P[i][1]+x[i]*(P[i][2]+x[i]*P[i][3])))
58  Ai.append(P[i][1]+x[i]*(2.*P[i][2]+x[i]*3.*P[i][3]))
59  Ai.append(P[i][2]+x[i]*3.*P[i][3])
60  Ai.append(P[i][3])
61  self.m_A.append(Ai)

64  return 0
