16 def ReferenceImpedance(S,Z0f,Z0i=None,Kf=None,Ki=None):
17  """Changes the reference impedance and scaling factor
18  @param S s-parameter matrix to convert
19  @param Z0f the reference impedance to convert to
20  @param Z0i (optional) the reference impedance of the s-parameters (assumed 50 Ohms)
21  @param Kf (optional) assumed to be sqrt(Z0f)
22  @param Ki (optional) assumed to be sqrt(Z0i)
23  @return the converted s-parameters
24  @see Z0KHelper to see how the reference impedance
25  and scaling factor are determined."""
26  (Z0f,Kf)=Z0KHelper((Z0f,Kf),len(S))
27  (Z0i,Ki)=Z0KHelper((Z0i,Ki),len(S))
28  I=matrix(identity(len(S)))
29  p=(matrix(Z0f)-matrix(Z0i))*(matrix(Z0f)+matrix(Z0i)).getI()
30  Kf=matrix(Ki)*matrix(Kf).getI()
31  S=matrix(S)
32  return (Kf*(I-p).getI()*(S-p)*(I-p*S).getI()*(I-p)*Kf.getI()).tolist()
